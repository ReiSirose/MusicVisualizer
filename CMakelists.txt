project(MusicVisualizer LANGUAGES CXX C)
cmake_minimum_required(VERSION 3.10) # 3.10 is a more compatible version

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

add_library(kissfft_lib STATIC
    third_party/kiss_fft/kiss_fft.c
    third_party/kiss_fft/kiss_fftr.c
)


target_include_directories(kissfft_lib PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/third_party/kiss_fft
)

add_executable(visualizer src/main.cpp)

# Add include paths for miniaudio.h and your own headers
target_include_directories(visualizer PRIVATE 
    ${CMAKE_CURRENT_SOURCE_DIR}/lib
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(visualizer PRIVATE 
    kissfft_lib  # This links libkissfft_lib.a
)

# Add platform-specific libraries (miniaudio needs these)
if(APPLE)
    target_link_libraries(visualizer PRIVATE 
        m 
        pthread 
        "-framework CoreFoundation" 
        "-framework CoreAudio"
    )
elseif(LINUX)
    target_link_libraries(visualizer PRIVATE m pthread dlfcn)
elseif(WIN32)
    target_link_libraries(visualizer PRIVATE) 
endif()

set_target_properties(visualizer PROPERTIES 
    CXX_EXTENSIONS OFF
)
